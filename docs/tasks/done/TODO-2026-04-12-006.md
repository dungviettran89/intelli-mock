## Route Testing Strategy
Route files export Express Router instances. Testing approach:
- Use supertest with a minimal Express app that uses the router
- Send requests to each registered route path
- Verify the request reaches the controller (mock controller with spy functions)
- Confirm HTTP method, path, and middleware chain are correct

Example pattern (from existing codebase conventions):
```ts
import request from 'supertest';
import express from 'express';
import { vi } from 'vitest';
import { mockRoutes } from '../../src/modules/mock/mock.routes.js';

// Mock the controller
vi.mock('../../src/modules/mock/mock.controller.js', () => ({
  MockController: vi.fn().mockImplementation(() => ({
    getAll: vi.fn((req, res) => res.json([])),
    getById: vi.fn((req, res) => res.json({})),
    // ... etc
  })),
}));

describe('MockRoutes', () => {
  let app: express.Application;

  beforeAll(() => {
    app = express();
    app.use(express.json());
    app.use('/api/mocks', mockRoutes);
  });

  it('should register GET /api/mocks', async () => {
    const res = await request(app).get('/api/mocks');
    expect(res.status).toBe(200);
  });
  // ... one test per route
});
```
